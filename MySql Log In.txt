create database if not exists BDLogIn;
use BDLogIn;
create table if not exists Perfil (
	Tipo varchar(50) NOT NULL,
	Id int NOT NULL,
    Primary Key (Id)
) Engine = InNoDB;

create table if not exists Usuarios (
	Nombre varchar(50),
    Apellido varchar(50),
    Username varchar(50),
    Contraseña varchar(50),
    Email varchar(50),
	Id int NOT NULL,
    Perfil int NOT NULL,
    Primary Key (Id)
) Engine = InNoDB;
